def truncated_normal(mean, y):
    """Used to sample z given beta
    needs a :
    - mean 
    - support borders where to sample
    """
    a = -mean
    b = -mean
    
    a[y<=0] = -np.inf
    b[y>0] = np.inf
    
    z = truncnorm.rvs(a, b, loc=mean, size=(len(mean),))
    
    return z

def Gibbs_Sampling(data, target, iterations, tau=100, burnin=3000):
    """Performs Gibbs sampling iteratively from z given beta,X,Y and beta given z,X,Y 
    to approximate samples from the join (z, beta) given (X,Y)
    returns : beta_burnin, beta_after_burnin, z_burnin, z_after_burnin
              Those are the samples in each direction (variable)
    """
    
    n = data.shape[0]
    dim = data.shape[1]
    sigma = np.linalg.inv((np.eye(dim) / tau ) + data.T @ data)
    used_to_speed_up_things = sigma @ data.T
    
    z = np.random.randn(n) # initialisation of z0 : a vector of size the training data sampled from the normal distribution
    list_z = []
    list_beta = []
    
    
    for t in range(burnin + iterations):
        mu = used_to_speed_up_things @ z
        beta = np.random.multivariate_normal(mu, cov=sigma) # beta given z sampling
        list_beta.append(beta)
        z = truncated_normal( data @ beta, target)
        list_z.append(z)
    
    return np.array(list_beta)[:burnin], np.array(list_beta)[burnin:], np.array(list_z)[:burnin], np.array(list_z)[burnin:]

def prediction_phase(data_test, betas_post_burnin):
    
    z_pred = data_test @ betas_post_burnin.T
    noise = np.random.randn(z_pred.shape[0], z_pred.shape[1])    
    return np.sign(z_pred+noise).replace(-1,0)

Mean Field variational sampling¶

We use the fact that the mean of a truncated gaussian is :

$$ X \beta + y \frac{\phi(X \beta)}{\Phi(yX\beta)}$$

where $\phi$ is the pdf of the normal distribution and $\Phi$ is the CDF

def mean_field(data, target, iterations, tau=100):
    
    n = data.shape[0]
    d = data.shape[1]
    sigma = np.linalg.inv(data.T @ data + (np.eye(d) / tau))
    
    betas = np.zeros((iterations, d))
    zs = np.zeros((iterations, n))
    beta = np.random.randn(d)[:,None] #initialization of beta
    
    for t in range(iterations):
        betas[t] = beta.ravel()
        mean = data @ beta.ravel()
        z =  mean + target * norm.pdf(mean) / norm.cdf(target * mean) #update z
        zs[t] = z
        beta = sigma @ (data.T @ z)#update beta
        
    return betas, zs

def sampling(mean_beta, mean_z, data, target, nb_samples, tau=100):
    
    n = data.shape[0]
    d = data.shape[1]
    sigma = np.linalg.inv(data.T @ data + np.eye(d) / tau)
    betas = np.random.multivariate_normal(mean=mean_beta,
                                          cov=sigma,
                                          size=nb_samples)
    z_s = np.zeros((nb_samples, n))
    for t in range(nb_samples):
        z_s[t] = truncated_normal(mean_z, target)
    return betas, z_s
